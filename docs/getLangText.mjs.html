/**
 * 基於指定語系取得指定鍵值
 *
 * @param {String} lang 輸入語系字串
 * @param {String} key 輸入指定鍵字串
 * @param {String} [def=''] 輸入若無法取得時之預設值字串，預設為''
 * @returns {String} 回傳值字串
 */
function getLangText(lang, key, def = '') {
    lang = lang.toLowerCase()
    let r = get(kp, `${lang}.${key}`, def)
    return r
}
